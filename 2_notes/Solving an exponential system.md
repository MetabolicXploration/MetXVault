$$\begin{align}
\frac{dx}{dt} &= rx
\\
\int_{0}^{t} d\tilde{t} &= \int_{x_0}^{x} \frac{\tilde{x}}{r\tilde{x}}
\\
t &= \frac{\log(rx)}{r} - \frac{\log(r x_0)}{r}
\\
rt &= \log(rx/rx_0)
\\
e^{rt} &= x/x_0
\\
x(t) &= x_0 e^{rt}
\end{align}$$
